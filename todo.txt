DELEGATE SUPPORT: Turning MethodInfo objects into Delegates at runtime, with absolutely no compile time type parameters/declarations
To create and invoke a delegate dynamically at runtime without knowing its signature at compile time, you can use Expression.GetDelegateType. This allows you to construct the exact delegate type dynamically based on the MethodInfo parameters and return type.Once you have the dynamic delegate type, you can use Delegate.CreateDelegate() and invoke it dynamically via its DynamicInvoke method.
using System;
using System.Linq;
using System.Linq.Expressions;
using System.Reflection;

public class DynamicRunner
{
    public static void Main()
    {
        // Example setup: We have a method name and an instance of an unknown object
        object targetObject = new Calculator();
        string methodName = "Multiply"; // Let's pretend this comes from user input/config
        object[] methodArguments = new object[] { 6, 7 }; // Arguments determined at runtime

        // 1. Reflect the method from the object's type
        MethodInfo methodInfo = targetObject.GetType().GetMethod(methodName);

        // 2. Dynamically build the delegate type signature using Expression Trees
        // Combine parameter types and the return type into an array
        Type[] parameterTypes = methodInfo.GetParameters().Select(p => p.ParameterType).ToArray();
        Type returnType = methodInfo.ReturnType;

        // Automatically resolves to the correct Func<> or Action<> type at runtime
        Type delegateType = Expression.GetDelegateType(parameterTypes.Concat(new[] { returnType }).ToArray());

        // 3. Create the strongly-typed delegate instance
        Delegate dynamicDelegate = Delegate.CreateDelegate(delegateType, targetObject, methodInfo);

        // 4. Invoke it dynamically
        object result = dynamicDelegate.DynamicInvoke(methodArguments);

        Console.WriteLine($"Result: {result}"); // Outputs: Result: 42
    }
}

// Dummy class for the example
public class Calculator
{
    public int Multiply(int a, int b) => a * b;
}
